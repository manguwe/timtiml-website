timtiml innovatives Widget

This standalone widget auto-injects into any timtiml innovatives page.
Click the chat button at the bottom-left to preview it.

<!-- Add to bottom of any page's <body> -->
<script>
(function(){
  // Optional config (before loading):
  window.CedricWidgetConfig = {
    greeting: "Hi! How can timtiml help you today?",
    faqHref: "#faq",
    agentRole: "Support Consultant"
  };
  var r = document.createElement('div');
  r.id = 'cedric-support-root';
  document.body.appendChild(r);
  fetch('support-widget.html')
    .then(function(res){ return res.text(); })
    .then(function(html){
      r.innerHTML = html;
      r.querySelectorAll('script:not([data-widget-meta])')
        .forEach(function(s){
          var n = document.createElement('script');
          n.textContent = s.textContent;
          document.body.appendChild(n);
        });
      var css = r.querySelector('#cedric-widget-styles');
      if(css){ document.head.appendChild(css.cloneNode(true)); }
    });
})();
</script>
